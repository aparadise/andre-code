1. global scope
2. the function scope 
            example 1:
            {
            var a = 'this a global block';
            }
            console.log(a);
            
            ----------------------------------
            example 2:
            
            function hello(){
             var a = 'this is function scope NOT global'
            }
            hello();
            console.log(a); //this will give error cuz the a inside of the function. the console.log() needs to be inside of the function for access.
